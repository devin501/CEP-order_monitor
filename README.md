## 五、配置说明
### 告警阈值
ALERT_CONFIG = {
    "normal_threshold": 5,      # 正常：< 5 条
    "warning_threshold": 10,    # 警告：5-10 条
    "critical_threshold": 20,   # 严重：> 10 条
    "consecutive_count": 3,     # 连续3次触发升级
}

### 报表时间配置

REPORT_CONFIG = {
    "daily_report_time": "09:00",    # 日报：每天 9:00
    "weekly_report_day": "Monday",   # 周报：周一
    "weekly_report_time": "10:00",   # 周报：10:00
}

### 环境变量
export FEISHU_WEBHOOK="https://open.feishu.cn/open-apis/bot/v2/hook/..."
export ORDER_API_TOKEN="bk-acb93eae-0112-4cff-af24-a3b476355839"

## 六、Cron 定时任务

# 每30分钟执行一次
*/30 * * * * export FEISHU_WEBHOOK="..." && export ORDER_API_TOKEN="..." && cd ~/.openclaw/workspace/scripts && /usr/bin/python3 order_monitor_pro.py >> order_monitor.log 2>&1

## 七、通知类型
场景	通知类型	内容
新增失败工单	普通通知	工单列表 + 统计
连续3次检查失败	🚨 紧急通知	@所有人
告警级别升级	🚨 紧急通知	@所有人
每天9:00	📊 日报	当日统计汇总
周一10:00	📆 周报	本周统计汇总
无新增工单	静默	不发送通知
